```rust
//! Daemon lifecycle management.
//!
//! Checks whether `ocsyncd` is already running (by attempting a connection),
//! and spawns it if not.

use std::path::Path;
use std::time::Duration;

use thiserror::Error;
use tokio::net::UnixStream;

/// Errors that can occur when ensuring the daemon is running.
#[derive(Debug, Error)]
pub enum SpawnError {
    /// The `ocsyncd` binary could not be found next to the `ocsync` binary.
    #[error("daemon binary 'ocsyncd' not found alongside ocsync executable")]
    NoBinary,

    /// The daemon could not be started or connected to after all retries.
    #[error("failed to connect to daemon: {0}")]
    Failed(String),
}

/// Try to connect to the GUI IPC socket up to `retries` times, with
/// `delay_ms` milliseconds between attempts.
///
/// Returns `Ok(())` on the first successful connection (the connection is
/// immediately dropped — this is just a liveness check).
/// Returns `Err(SpawnError::Failed)` if all attempts fail.
pub async fn wait_for_socket(
    socket_path: &Path,
    retries: u32,
    delay_ms: u64,
) -> Result<(), SpawnError> {
    for attempt in 0..retries {
        match UnixStream::connect(socket_path).await {
            Ok(_stream) => return Ok(()),
            Err(e) => {
                tracing::debug!(
                    "connect attempt {}/{}: {}",
                    attempt + 1,
                    retries,
                    e
                );
                tokio::time::sleep(Duration::from_millis(delay_ms)).await;
            }
        }
    }
    Err(SpawnError::Failed(format!(
        "socket {} not reachable after {retries} attempts",
        socket_path.display()
    )))
}

/// Ensure the `ocsyncd` daemon is running.
///
/// Algorithm:
/// 1. Try to connect to `socket_path`.
/// 2. If connection succeeds, return `Ok(())` — daemon already running.
/// 3. If connection fails, locate the `ocsyncd` binary in the same directory
///    as the current `ocsync` binary (`std::env::current_exe()`).
/// 4. If binary not found, return `Err(SpawnError::NoBinary)`.
/// 5. Spawn the daemon with `std::process::Command`.
/// 6. Retry the connection up to 5 times with 200 ms delay.
/// 7. Return `Ok(())` on success, `Err(SpawnError::Failed)` if all retries fail.
pub async fn ensure_daemon_running(socket_path: &Path) -> Result<(), SpawnError> {
    // Fast path: daemon already running.
    if UnixStream::connect(socket_path).await.is_ok() {
        return Ok(());
    }

    // Locate daemon binary next to our own executable.
    let daemon_path = find_daemon_binary()?;

    tracing::info!("spawning daemon: {}", daemon_path.display());

    std::process::Command::new(&daemon_path)
        .spawn()
        .map_err(|e| SpawnError::Failed(format!("failed to spawn daemon: {e}")))?;

    // Retry connect.
    wait_for_socket(socket_path, 5, 200).await
}

/// Find the `ocsyncd` binary in the same directory as the running `ocsync`
/// executable. Returns `Err(SpawnError::NoBinary)` if not found.
fn find_daemon_binary() -> Result<std::path::PathBuf, SpawnError> {
    let exe = std::env::current_exe()
        .map_err(|e| SpawnError::Failed(format!("cannot determine current exe: {e}")))?;

    let dir = exe
        .parent()
        .ok_or_else(|| SpawnError::Failed("exe has no parent directory".to_string()))?;

    let candidate = dir.join("ocsyncd");

    if candidate.exists() {
        Ok(candidate)
    } else {
        Err(SpawnError::NoBinary)
    }
}
```

//! IPC connection to the `ocsyncd` daemon.
//!
//! The GUI connects to the daemon's Unix socket, sends JSON-newline
//! `DaemonCommand` messages, and reads back JSON-newline `DaemonEvent`
//! messages on a background reader task.

use std::path::Path;

use thiserror::Error;
use tokio::io::{AsyncBufReadExt, AsyncWriteExt, BufReader};
use tokio::net::UnixStream;
use tokio::sync::mpsc;

use daemon::gui_ipc::protocol::{DaemonCommand, DaemonEvent};

/// Errors that can occur while connecting to the daemon.
#[derive(Debug, Error)]
pub enum ConnError {
    #[error("I/O error: {0}")]
    Io(#[from] std::io::Error),

    #[error("JSON error: {0}")]
    Json(#[from] serde_json::Error),
}

/// A connected (or intentionally disconnected) handle to the daemon.
///
/// Commands are sent fire-and-forget via an internal `mpsc::Sender`.
/// A background task forwards them to the socket and reads events.
#[derive(Debug, Clone)]
pub struct DaemonConnection {
    cmd_tx: mpsc::Sender<DaemonCommand>,
}

impl DaemonConnection {
    /// Return a stub connection whose send channel is immediately closed.
    ///
    /// Calling `send()` on a disconnected connection silently drops commands.
    pub fn disconnected() -> Self {
        let (tx, _rx) = mpsc::channel(1);
        Self { cmd_tx: tx }
    }

    /// Connect to the daemon Unix socket at `socket_path`.
    ///
    /// Spawns two background tasks:
    /// - **Writer task**: drains the internal command channel and writes
    ///   newline-delimited JSON to the socket.
    /// - **Reader task**: reads newline-delimited JSON events from the socket
    ///   and forwards them to the returned `mpsc::Receiver<DaemonEvent>`.
    ///
    /// Returns `(DaemonConnection, event_rx)`.
    pub async fn connect(
        socket_path: &Path,
    ) -> Result<(Self, mpsc::Receiver<DaemonEvent>), ConnError> {
        let stream = UnixStream::connect(socket_path).await?;
        let (read_half, mut write_half) = stream.into_split();

        let (cmd_tx, mut cmd_rx) = mpsc::channel::<DaemonCommand>(64);
        let (event_tx, event_rx) = mpsc::channel::<DaemonEvent>(64);

        // Writer task: receive DaemonCommand messages and send to daemon.
        tokio::spawn(async move {
            while let Some(cmd) = cmd_rx.recv().await {
                match serde_json::to_string(&cmd) {
                    Ok(json) => {
                        let line = json + "\n";
                        if write_half.write_all(line.as_bytes()).await.is_err() {
                            tracing::warn!("daemon write half closed");
                            break;
                        }
                    }
                    Err(e) => {
                        tracing::error!("failed to serialize command: {e}");
                    }
                }
            }
        });

        // Reader task: read DaemonEvent JSON lines and forward to event_tx.
        tokio::spawn(async move {
            let mut reader = BufReader::new(read_half);
            let mut line = String::new();
            loop {
                line.clear();
                match reader.read_line(&mut line).await {
                    Ok(0) => {
                        tracing::info!("daemon closed connection (EOF)");
                        break;
                    }
                    Ok(_) => {
                        let trimmed = line.trim();
                        if trimmed.is_empty() {
                            continue;
                        }
                        match serde_json::from_str::<DaemonEvent>(trimmed) {
                            Ok(event) => {
                                if event_tx.send(event).await.is_err() {
                                    // GUI dropped the receiver — stop reading.
                                    break;
                                }
                            }
                            Err(e) => {
                                tracing::warn!("failed to parse daemon event: {e} — line: {trimmed}");
                            }
                        }
                    }
                    Err(e) => {
                        tracing::warn!("daemon read error: {e}");
                        break;
                    }
                }
            }
        });

        Ok((Self { cmd_tx }, event_rx))
    }

    /// Send a command to the daemon (fire-and-forget).
    ///
    /// If the connection is closed or the channel is full the command is
    /// silently dropped (errors are logged at WARN level).
    pub fn send(&self, cmd: DaemonCommand) {
        if let Err(e) = self.cmd_tx.try_send(cmd) {
            tracing::warn!("failed to send daemon command: {e}");
        }
    }
}

//! iced subscription that drives the DaemonEvent stream into the update loop.
//!
//! `daemon_events` creates an `iced::Subscription` that reads from an
//! `mpsc::Receiver<DaemonEvent>` and yields `Message::DaemonEvent(e)` for
//! each event, then `Message::DaemonDisconnected` when the channel closes.

use tokio::sync::mpsc;

use daemon::gui_ipc::protocol::DaemonEvent;

use crate::app::Message;

/// Pull the next item from the receiver and convert it to a `Message`.
///
/// Returns `None` only after `DaemonDisconnected` has been returned once
/// (so the subscription can yield "nothing forever" afterward).
///
/// Exposed as `pub` so unit tests can drive the state machine directly.
pub async fn next_message(rx: &mut mpsc::Receiver<DaemonEvent>) -> Option<Message> {
    match rx.recv().await {
        Some(event) => Some(Message::DaemonEvent(event)),
        None => Some(Message::DaemonDisconnected),
    }
}

/// Create an iced `Subscription` that yields `Message` values from the daemon
/// event channel.
///
/// Uses `iced::subscription::unfold` with:
/// - **State** = `Option<mpsc::Receiver<DaemonEvent>>`
///   - `Some(rx)` → still reading events
///   - `None` → disconnected, yield nothing forever
/// - Each `Some(rx)` iteration: call `rx.recv()` →
///   - `Some(event)` → emit `Message::DaemonEvent(event)`, stay in `Some(rx)`
///   - `None` (closed) → emit `Message::DaemonDisconnected`, move to `None`
/// - `None` state: the subscription idles (sleeps forever).
pub fn daemon_events(event_rx: mpsc::Receiver<DaemonEvent>) -> iced::Subscription<Message> {
    iced::subscription::unfold(
        std::any::TypeId::of::<DaemonEvent>(),
        Some(event_rx),
        |state| async move {
            match state {
                Some(mut rx) => match rx.recv().await {
                    Some(event) => (Message::DaemonEvent(event), Some(rx)),
                    None => (Message::DaemonDisconnected, None),
                },
                None => {
                    // Channel closed — idle forever.
                    let () = std::future::pending().await;
                    unreachable!()
                }
            }
        },
    )
}

use uuid::Uuid;

use daemon::gui_ipc::protocol::{DaemonCommand, DaemonEvent};

            tray: None,

/// All messages that can be dispatched to `update()`.
#[derive(Debug, Clone)]
pub enum Message {
    /// An event received from the daemon IPC stream.
    DaemonEvent(DaemonEvent),
    /// The daemon IPC channel has closed (daemon exited or connection lost).
    DaemonDisconnected,
    /// A tray icon event (click, menu item, etc.).
    TrayEvent(TrayIconEventWrapper),
    /// Navigate to a different view.
    NavigateTo(View),
    /// Toggle the main window visibility.
    ToggleWindow,
    /// User typed in the Add Account URL field.
    AddAccountUrlChanged(String),
    /// User submitted the Add Account form.
    AddAccountSubmit,
    /// Pause sync for a folder.
    PauseFolder(Uuid),
    /// Resume sync for a folder.
    ResumeFolder(Uuid),
    /// Force an immediate sync for a folder.
    ForceSyncFolder(Uuid),
    /// Remove an account.
    RemoveAccount(Uuid),
    /// Quit the application.
    Quit,
}

/// Thin wrapper around tray-icon events so `Message` can derive Clone.
#[derive(Debug, Clone)]
pub enum TrayIconEventWrapper {
    LeftClick,
    RightClick,
    MenuItemClick(String),
}

/// The core state-transition function.
///
/// Mutates `app` in-place and returns an `iced::Command<Message>` for any
/// async side-effects (window show/hide, quit).
pub fn update(app: &mut App, message: Message) -> iced::Command<Message> {
    match message {
        Message::DaemonEvent(event) => handle_daemon_event(app, event),

        Message::DaemonDisconnected => {
            tracing::warn!("daemon disconnected");
            iced::Command::none()
        }

        Message::TrayEvent(event) => match event {
            TrayIconEventWrapper::LeftClick => {
                app.window_visible = !app.window_visible;
                set_window_visibility(app.window_visible)
            }
            TrayIconEventWrapper::RightClick => iced::Command::none(),
            TrayIconEventWrapper::MenuItemClick(id) => match id.as_str() {
                "quit" => {
                    app.daemon.send(DaemonCommand::Quit);
                    iced::window::close(iced::window::Id::MAIN)
                }
                "open" => {
                    app.window_visible = true;
                    set_window_visibility(true)
                }
                _ => iced::Command::none(),
            },
        },

        Message::NavigateTo(view) => {
            app.active_view = view;
            iced::Command::none()
        }

        Message::ToggleWindow => {
            app.window_visible = !app.window_visible;
            set_window_visibility(app.window_visible)
        }

        Message::AddAccountUrlChanged(url) => {
            if let View::AddAccount { url_input, .. } = &mut app.active_view {
                *url_input = url;
            }
            iced::Command::none()
        }

        Message::AddAccountSubmit => {
            if let View::AddAccount { url_input, error } = &mut app.active_view {
                let url = url_input.clone();
                if url.is_empty() {
                    *error = Some("Please enter a server URL".to_string());
                } else {
                    *error = None;
                    app.daemon.send(DaemonCommand::AddAccount { url });
                }
            }
            iced::Command::none()
        }

        Message::PauseFolder(folder_id) => {
            app.daemon.send(DaemonCommand::PauseFolder { folder_id });
            set_folder_status(app, folder_id, FolderStatus::Paused);
            iced::Command::none()
        }

        Message::ResumeFolder(folder_id) => {
            app.daemon.send(DaemonCommand::ResumeFolder { folder_id });
            set_folder_status(app, folder_id, FolderStatus::Idle);
            iced::Command::none()
        }

        Message::ForceSyncFolder(folder_id) => {
            app.daemon.send(DaemonCommand::TriggerSync { folder_id });
            iced::Command::none()
        }

        Message::RemoveAccount(account_id) => {
            app.daemon
                .send(DaemonCommand::RemoveAccount { account_id });
            app.accounts.retain(|a| a.id != account_id);
            app.active_view = View::SyncStatus;
            iced::Command::none()
        }

        Message::Quit => {
            app.daemon.send(DaemonCommand::Quit);
            iced::window::close(iced::window::Id::MAIN)
        }
    }
}

// ── helpers ───────────────────────────────────────────────────────────────────

fn handle_daemon_event(app: &mut App, event: DaemonEvent) -> iced::Command<Message> {
    match event {
        DaemonEvent::Ready => {
            // Daemon is ready; no state change needed.
        }

        DaemonEvent::SyncStarted { folder_id } => {
            if let Some(folder) = find_folder_mut(app, folder_id) {
                folder.status = FolderStatus::Syncing;
                folder.errors.clear();
            }
        }

        DaemonEvent::SyncProgress {
            folder_id,
            done,
            total,
        } => {
            if let Some(folder) = find_folder_mut(app, folder_id) {
                folder.progress = Some((done, total));
            }
        }

        DaemonEvent::SyncFinished { folder_id, errors } => {
            if let Some(folder) = find_folder_mut(app, folder_id) {
                folder.progress = None;
                if errors.is_empty() {
                    folder.status = FolderStatus::Idle;
                    folder.errors.clear();
                } else {
                    folder.status = FolderStatus::Error;
                    folder.errors = errors;
                }
            }
        }

        DaemonEvent::FileStatusChanged { path, status } => {
            tracing::debug!("file status changed: {path} → {status}");
        }

        DaemonEvent::AccountStateChanged { account_id, state } => {
            tracing::debug!("account state changed: {account_id} → {state}");
        }
    }

    iced::Command::none()
}

fn find_folder_mut<'a>(app: &'a mut App, folder_id: Uuid) -> Option<&'a mut FolderView> {
    app.accounts
        .iter_mut()
        .flat_map(|a| a.folders.iter_mut())
        .find(|f| f.id == folder_id)
}

fn set_folder_status(app: &mut App, folder_id: Uuid, status: FolderStatus) {
    if let Some(folder) = find_folder_mut(app, folder_id) {
        folder.status = status;
    }
}

fn set_window_visibility(visible: bool) -> iced::Command<Message> {
    if visible {
        iced::window::change_mode(iced::window::Id::MAIN, iced::window::Mode::Windowed)
    } else {
        iced::window::change_mode(iced::window::Id::MAIN, iced::window::Mode::Hidden)
    }
}

//! Tray icon setup and event handling.
//!
//! Platform notes:
//! - macOS: Set `LSUIElement = true` in `Info.plist` to suppress the Dock icon
//!   when the window is hidden. This is done in the app bundle build step.
//! - All platforms: left-click → toggle window; right-click → context menu.

use tokio::sync::mpsc;
use tray_icon::{
    menu::{Menu, MenuEvent, MenuItem, PredefinedMenuItem},
    TrayIcon, TrayIconBuilder, TrayIconEvent,
};

use crate::app::{Message, TrayIconEventWrapper};

/// Opaque handle to the tray icon.
///
/// The tray icon is kept alive as long as this handle exists; dropping it
/// removes the icon from the system tray.
#[derive(Debug)]
pub struct TrayHandle {
    _icon: TrayIcon,
}

// Implement Clone as a no-op (the handle is not truly cloneable, but App
// derives Clone and TrayHandle is behind Option<TrayHandle>).
impl Clone for TrayHandle {
    fn clone(&self) -> Self {
        panic!("TrayHandle must not be cloned; use Option<Arc<TrayHandle>> for shared ownership")
    }
}

/// Create and register the system tray icon.
///
/// Returns `(TrayHandle, event_rx)` where `event_rx` delivers raw
/// `TrayIconEvent` values. The caller should convert these to `Message` via
/// [`tray_event_to_message`] and inject them into the iced update loop.
pub fn setup_tray() -> Result<(TrayHandle, mpsc::Receiver<Message>), tray_icon::Error> {
    // Build a 16×16 RGBA placeholder icon (solid blue square).
    // Replace with actual icon asset in production.
    let icon_rgba: Vec<u8> = [0u8, 100, 200, 255].repeat(16 * 16);
    let icon = tray_icon::Icon::from_rgba(icon_rgba, 16, 16)?;

    // Build the right-click context menu.
    let open_item = MenuItem::with_id("open", "Open ownCloud", true, None);
    let quit_item = MenuItem::with_id("quit", "Quit", true, None);

    let menu = Menu::new();
    menu.append(&open_item)?;
    menu.append(&PredefinedMenuItem::separator())?;
    menu.append(&quit_item)?;

    let tray_icon = TrayIconBuilder::new()
        .with_tooltip("ownCloud")
        .with_icon(icon)
        .with_menu(Box::new(menu))
        .build()?;

    let (tx, rx) = mpsc::channel::<Message>(32);

    // Forward TrayIconEvent → Message via a background thread.
    // tray-icon uses a callback model; we bridge it to a tokio channel.
    let tx_clone = tx.clone();
    TrayIconEvent::set_event_handler(Some(move |event: TrayIconEvent| {
        let msg = match event {
            TrayIconEvent::Click {
                button: tray_icon::MouseButton::Left,
                ..
            } => Message::TrayEvent(TrayIconEventWrapper::LeftClick),
            TrayIconEvent::Click {
                button: tray_icon::MouseButton::Right,
                ..
            } => Message::TrayEvent(TrayIconEventWrapper::RightClick),
            _ => return,
        };
        let _ = tx_clone.try_send(msg);
    }));

    // Forward MenuEvent → Message.
    MenuEvent::set_event_handler(Some(move |event: MenuEvent| {
        let msg = Message::TrayEvent(TrayIconEventWrapper::MenuItemClick(event.id.0.clone()));
        let _ = tx.try_send(msg);
    }));

    Ok((TrayHandle { _icon: tray_icon }, rx))
}

/// Returns the platform-specific GUI IPC socket path.

// ── iced Application wrapper ──────────────────────────────────────────────────

struct OcsyncApp {
    state: App,
    event_rx: Arc<tokio::sync::Mutex<mpsc::Receiver<daemon::gui_ipc::protocol::DaemonEvent>>>,
    tray_rx: Arc<tokio::sync::Mutex<mpsc::Receiver<Message>>>,
}

impl Application for OcsyncApp {
    type Executor = iced::executor::Default;
    type Message = Message;
    type Theme = iced::Theme;
    type Flags = (
        App,
        mpsc::Receiver<daemon::gui_ipc::protocol::DaemonEvent>,
        mpsc::Receiver<Message>,
    );

    fn new(flags: Self::Flags) -> (Self, iced::Command<Message>) {
        let (state, event_rx, tray_rx) = flags;
        (
            Self {
                state,
                event_rx: Arc::new(tokio::sync::Mutex::new(event_rx)),
                tray_rx: Arc::new(tokio::sync::Mutex::new(tray_rx)),
            },
            iced::Command::none(),
        )
    }

    fn title(&self) -> String {
        "ownCloud".to_string()
    }

    fn update(&mut self, message: Message) -> iced::Command<Message> {
        app::update(&mut self.state, message)
    }

    fn view(&self) -> iced::Element<Message> {
        match &self.state.active_view {
            View::SyncStatus => sync_status_view(&self.state.accounts),
            View::AccountSettings(account_id) => {
                let account_id = *account_id;
                if let Some(account) = self.state.accounts.iter().find(|a| a.id == account_id) {
                    account_settings_view(account)
                } else {
                    sync_status_view(&self.state.accounts)
                }
            }
            View::AddAccount { url_input, error } => {
                add_account_view(url_input, error.as_deref())
            }
            View::GeneralSettings => general_settings_view(),
        }
    }

    fn subscription(&self) -> iced::Subscription<Message> {
        let event_rx_ref = self.event_rx.clone();
        let tray_rx_ref = self.tray_rx.clone();

        // We create subscriptions via unfold using the Arc<Mutex<Receiver>>
        // to avoid moving the receivers out of self.
        let daemon_sub = iced::subscription::unfold(
            "daemon-events",
            event_rx_ref,
            |rx| async move {
                let mut guard = rx.lock().await;
                let msg = match guard.recv().await {
                    Some(event) => Message::DaemonEvent(event),
                    None => Message::DaemonDisconnected,
                };
                drop(guard);
                (msg, rx)
            },
        );

        let tray_sub = iced::subscription::unfold(
            "tray-events",
            tray_rx_ref,
            |rx| async move {
                let mut guard = rx.lock().await;
                // Idle if no tray events are available.
                let msg = match guard.recv().await {
                    Some(msg) => msg,
                    None => {
                        // Channel closed; sleep forever.
                        let () = std::future::pending().await;
                        unreachable!()
                    }
                };
                drop(guard);
                (msg, rx)
            },
        );

        iced::Subscription::batch([daemon_sub, tray_sub])
    }
}
